 25async def first(xs: typing.AsyncIterable[T],
 26                fn: typing.Callable[[T], typing.Any] = lambda _: True,
 27                default: typing.Optional[T] = None
 28                ) -> typing.Optional[T]:
 29    """Return the first element from async iterable that satisfies
 30    predicate `fn`, or `default` if no such element exists.
 31
 32    Result of predicate `fn` can be of any type. Predicate is satisfied if it's
 33    return value is truthy.
 34
 35    Args:
 36        xs: async collection
 37        fn: predicate
 38        default: default value
 39
 40    Example::
 41
 42        async def async_range(x):
 43            for i in range(x):
 44                await asyncio.sleep(0)
 45                yield i
 46
 47        assert await first(async_range(3)) == 0
 48        assert await first(async_range(3), lambda x: x > 1) == 2
 49        assert await first(async_range(3), lambda x: x > 2) is None
 50        assert await first(async_range(3), lambda x: x > 2, 123) == 123
 51
 52    """
 53    async for i in xs:
 54        if fn(i):
 55            return i
 56    return default

 59async def uncancellable(f: asyncio.Future,
 60                        raise_cancel: bool = True
 61                        ) -> typing.Any:
 62    """Uncancellable execution of a Future.
 63
 64    Future is shielded and its execution cannot be interrupted.
 65
 66    If `raise_cancel` is `True` and the Future gets canceled,
 67    `asyncio.CancelledError` is reraised after the Future finishes.
 68
 69    Warning:
 70        If `raise_cancel` is `False`, this method suppresses
 71        `asyncio.CancelledError` and stops its propagation. Use with
 72        caution.
 73
 74    Args:
 75        f: future
 76        raise_cancel: raise CancelledError flag
 77
 78    Returns:
 79        future's result
 80
 81    """
 82    exception = None
 83    task = asyncio.ensure_future(f)
 84    while not task.done():
 85        try:
 86            await asyncio.shield(task)
 87        except asyncio.CancelledError as e:
 88            if raise_cancel:
 89                exception = e
 90        except Exception:
 91            pass
 92    if exception:
 93        raise exception
 94    return task.result()

248async def wait_for(f: typing.Awaitable,
249                   timeout: float
250                   ) -> typing.Any:
251    """"Wait for the single Future or coroutine to complete, with timeout.
252
253    Implementation `asyncio.wait_for` that ensure propagation of
254    CancelledError.
255
256    If task is cancelled with future's result available, instead of
257    returning result, this implementation raises `CancelledWithResultError`.
258
259    """
260    group = Group(exception_cb=lambda e: None)
261    group.spawn(call_on_done, asyncio.sleep(timeout), group.close)
262
263    f = group.wrap(f)
264    group.spawn(call_on_done, asyncio.shield(f), group.close)
265
266    exc = None
267
268    try:
269        await group.wait_closing()
270    except asyncio.CancelledError as e:
271        exc = e
272
273    try:
274        await uncancellable(group.async_close())
275    except asyncio.CancelledError as e:
276        exc = e
277
278    if exc:
279        if f.cancelled():
280            raise exc
281        result = None if f.exception() else f.result()
282        raise CancelledWithResultError(result, f.exception()) from exc
283
284    if f.cancelled():
285        raise asyncio.TimeoutError()
286
287    return f.result()

366def run_asyncio(future: typing.Awaitable, *,
367                handle_signals: bool = True,
368                loop: typing.Optional[asyncio.AbstractEventLoop] = None
369                ) -> typing.Any:
370    """Run asyncio loop until the `future` is completed and return the result.
371
372    If `handle_signals` is ``True``, SIGINT and SIGTERM handlers are
373    temporarily overridden. Instead of raising ``KeyboardInterrupt`` on every
374    signal reception, Future is canceled only once. Additional signals are
375    ignored. On Windows, SIGBREAK (CTRL_BREAK_EVENT) handler is also
376    overridden.
377
378    If `loop` is set to ``None``, new event loop is created and set
379    as thread's default event loop.
380
381    On Windows, asyncio loop gets periodically woken up (every 0.5 seconds).
382
383    Args:
384        future: future or coroutine
385        handle_signals: handle signals flag
386        loop: event loop
387
388    Returns:
389        future's result
390
391    Example::
392
393        async def run():
394            await asyncio.sleep(0)
395            return 123
396
397        result = run_asyncio(run())
398        assert result == 123
399
400    """
401    if not loop:
402        loop = asyncio.new_event_loop()
403        asyncio.set_event_loop(loop)
404
405    task = asyncio.ensure_future(future, loop=loop)
406
407    if sys.platform == 'win32':
408
409        async def task_wrapper(task):
410            try:
411                while not task.done():
412                    await asyncio.wait([task], timeout=0.5)
413            except asyncio.CancelledError:
414                task.cancel()
415            return await task
416
417        task = asyncio.ensure_future(task_wrapper(task), loop=loop)
418
419    if not handle_signals:
420        return loop.run_until_complete(task)
421
422    canceled = False
423    signalnums = [signal.SIGINT, signal.SIGTERM]
424    if sys.platform == 'win32':
425        signalnums += [signal.SIGBREAK]
426
427    def signal_handler(*args):
428        nonlocal canceled
429        if canceled:
430            return
431        loop.call_soon_threadsafe(task.cancel)
432        canceled = True
433
434    handlers = {signalnum: signal.getsignal(signalnum) or signal.SIG_DFL
435                for signalnum in signalnums}
436    for signalnum in signalnums:
437        signal.signal(signalnum, signal_handler)
438
439    try:
440        return loop.run_until_complete(task)
441
442    finally:
443        for signalnum, handler in handlers.items():
444            signal.signal(signalnum, handler)

459class Queue:
460    """Asyncio queue which implements AsyncIterable and can be closed.
461
462    Interface and implementation are based on `asyncio.Queue`.
463
464    If `maxsize` is less than or equal to zero, the queue size is infinite.
465
466    Args:
467        maxsize: maximum number of items in the queue
468
469    Example::
470
471        queue = Queue(maxsize=1)
472
473        async def producer():
474            for i in range(4):
475                await queue.put(i)
476            queue.close()
477
478        async def consumer():
479            result = 0
480            async for i in queue:
481                result += i
482            return result
483
484        asyncio.ensure_future(producer())
485        result = await consumer()
486        assert result == 6
487
488    """
489
490    def __init__(self,
491                 maxsize: int = 0, *,
492                 loop: typing.Optional[asyncio.AbstractEventLoop] = None):
493        self._maxsize = maxsize
494        self._loop = loop
495        self._queue = collections.deque()
496        self._getters = collections.deque()
497        self._putters = collections.deque()
498        self._closed = asyncio.Future(loop=loop)
499
500    def __aiter__(self):
501        return self
502
503    async def __anext__(self):
504        try:
505            return await self.get()
506        except QueueClosedError:
507            raise StopAsyncIteration
508
509    def __str__(self):
510        return (f'<{type(self).__name__}'
511                f' _closed={self._closed.done()} '
512                f' _queue={list(self._queue)}>')
513
514    def __len__(self):
515        return len(self._queue)
516
517    @property
518    def maxsize(self) -> int:
519        """Maximum number of items in the queue."""
520        return self._maxsize
521
522    @property
523    def is_closed(self) -> bool:
524        """Is queue closed."""
525        return self._closed.done()
526
527    def empty(self) -> bool:
528        """``True`` if queue is empty, ``False`` otherwise."""
529        return not self._queue
530
531    def full(self) -> bool:
532        """``True`` if queue is full, ``False`` otherwise."""
533        return (len(self._queue) >= self._maxsize if self._maxsize > 0
534                else False)
535
536    def qsize(self) -> int:
537        """Number of items currently in the queue."""
538        return len(self._queue)
539
540    def close(self):
541        """Close the queue."""
542        if self._closed.done():
543            return
544        self._closed.set_result(True)
545        self._wakeup_all(self._putters)
546        self._wakeup_next(self._getters)
547
548    def get_nowait(self) -> typing.Any:
549        """Return an item if one is immediately available, else raise
550        `QueueEmptyError`.
551
552        Raises:
553            QueueEmptyError
554
555        """
556        if self.empty():
557            raise QueueEmptyError()
558        item = self._queue.popleft()
559        self._wakeup_next(self._putters)
560        return item
561
562    def put_nowait(self, item: typing.Any):
563        """Put an item into the queue without blocking.
564
565        If no free slot is immediately available, raise `QueueFullError`.
566
567        Raises:
568            QueueFullError
569
570        """
571        if self._closed.done():
572            raise QueueClosedError()
573        if self.full():
574            raise QueueFullError()
575        self._queue.append(item)
576        self._wakeup_next(self._getters)
577
578    async def get(self) -> typing.Any:
579        """Remove and return an item from the queue.
580
581        If queue is empty, wait until an item is available.
582
583        Raises:
584            QueueClosedError
585
586        """
587        while self.empty():
588            if self._closed.done():
589                self._wakeup_all(self._getters)
590                raise QueueClosedError()
591            getter = asyncio.Future()
592            self._getters.append(getter)
593            try:
594                await getter
595            except BaseException:
596                getter.cancel()
597                with contextlib.suppress(ValueError):
598                    self._getters.remove(getter)
599                if not getter.cancelled():
600                    if not self.empty() or self._closed.done():
601                        self._wakeup_next(self._getters)
602                raise
603        return self.get_nowait()
604
605    async def put(self, item: typing.Any):
606        """Put an item into the queue.
607
608        If the queue is full, wait until a free slot is available before adding
609        the item.
610
611        Raises:
612            QueueClosedError
613
614        """
615        while not self._closed.done() and self.full():
616            putter = asyncio.Future()
617            self._putters.append(putter)
618            try:
619                await putter
620            except BaseException:
621                putter.cancel()
622                with contextlib.suppress(ValueError):
623                    self._putters.remove(putter)
624                if not self.full() and not putter.cancelled():
625                    self._wakeup_next(self._putters)
626                raise
627        return self.put_nowait(item)
628
629    async def get_until_empty(self) -> typing.Any:
630        """Empty the queue and return the last item.
631
632        If queue is empty, wait until at least one item is available.
633
634        Raises:
635            QueueClosedError
636
637        """
638        item = await self.get()
639        while not self.empty():
640            item = self.get_nowait()
641        return item
642
643    def get_nowait_until_empty(self) -> typing.Any:
644        """Empty the queue and return the last item if at least one
645        item is immediately available, else raise `QueueEmptyError`.
646
647        Raises:
648            QueueEmptyError
649
650        """
651        item = self.get_nowait()
652        while not self.empty():
653            item = self.get_nowait()
654        return item
655
656    def _wakeup_next(self, waiters):
657        while waiters:
658            waiter = waiters.popleft()
659            if not waiter.done():
660                waiter.set_result(None)
661                break
662
663    def _wakeup_all(self, waiters):
664        while waiters:
665            waiter = waiters.popleft()
666            if not waiter.done():
667                waiter.set_result(None)

674class Group:
675    """Group of asyncio Tasks.
676
677    Group enables creation and management of related asyncio Tasks. The
678    Group ensures uninterrupted execution of Tasks and Task completion upon
679    Group closing.
680
681    Group can contain subgroups, which are independent Groups managed by the
682    parent Group.
683
684    If a Task raises exception, other Tasks continue to execute.
685
686    If `exception_cb` handler is ``None``, exceptions are logged with level
687    WARNING.
688
689    """
690
691    def __init__(self,
692                 exception_cb: typing.Optional[ExceptionCb] = None,
693                 *,
694                 loop: typing.Optional[asyncio.AbstractEventLoop] = None):
695        self._exception_cb = exception_cb
696        self._loop = loop or asyncio.get_running_loop()
697        self._closing = asyncio.Future(loop=loop)
698        self._closed = asyncio.Future(loop=loop)
699        self._canceled = False
700        self._tasks = set()
701        self._parent = None
702        self._children = set()
703
704    @property
705    def is_open(self) -> bool:
706        """``True`` if group is not closing or closed, ``False`` otherwise."""
707        return not self._closing.done()
708
709    @property
710    def is_closing(self) -> bool:
711        """Is group closing or closed."""
712        return self._closing.done()
713
714    @property
715    def is_closed(self) -> bool:
716        """Is group closed."""
717        return self._closed.done()
718
719    async def wait_closing(self):
720        """Wait until closing is ``True``."""
721        await asyncio.shield(self._closing)
722
723    async def wait_closed(self):
724        """Wait until closed is ``True``."""
725        await asyncio.shield(self._closed)
726
727    def create_subgroup(self) -> 'Group':
728        """Create new Group as a child of this Group. Return the new Group.
729
730        When a parent Group gets closed, all of its children are closed.
731        Closing of a subgroup has no effect on the parent Group.
732
733        Subgroup inherits exception handler from its parent.
734
735        """
736        if self._closing.done():
737            raise Exception('group not open')
738        child = Group(self._exception_cb, loop=self._loop)
739        child._parent = self
740        self._children.add(child)
741        return child

743    def wrap(self,
744             future: asyncio.Future
745             ) -> asyncio.Task:
746        """Wrap the Future into a Task and schedule its execution. Return the
747        Task object.
748
749        Resulting task is shielded and can be canceled only with
750        `Group.async_close`.
751
752        """
753        if self._closing.done():
754            raise Exception('group not open')
755        task = asyncio.ensure_future(future, loop=self._loop)
756        self._tasks.add(task)
757        task.add_done_callback(self._on_task_done)
758        return asyncio.shield(task)

760    def spawn(self,
761              fn: typing.Callable[..., typing.Awaitable],
762              *args, **kwargs
763              ) -> asyncio.Task:
764        """Wrap the result of a `fn` into a Task and schedule its execution.
765        Return the Task object.
766
767        Function `fn` is called with provided `args` and `kwargs`.
768        Resulting Task is shielded and can be canceled only with
769        `Group.async_close`.
770
771        """
772        if self._closing.done():
773            raise Exception('group not open')
774        future = fn(*args, **kwargs)
775        return self.wrap(future)
776
777    def close(self, cancel: bool = True):
778        """Schedule Group closing.
779
780        Closing Future is set immediately. All subgroups are closed, and all
781        running tasks are optionally canceled. Once closing of all subgroups
782        and execution of all tasks is completed, closed Future is set.
783
784        Tasks are canceled if `cancel` is ``True``.
785
786        """
787        for child in list(self._children):
788            child.close(cancel)
789        if cancel and not self._canceled:
